Python es modular y tiene espacios de nombre

  • Un .py es un módulo.

Clases

# posicion.py

import math     # baterias incluídas!

class Posicion(object):

    def __init__(self, x, y):
        self.x = x
        self.y = y

    def distancia(self):
        x = self.x**2 + self.y**2
        return math.sqrt(x)
